#include<iostream>
#include<vector>
using namespace std;
//快排-->递归，基准元素
void Qsort(vector<int> &x,int l,int r)
{
    int i=l,j=r,p=x[l];
    //至少两个元素
    while(i<j)
    {
        //从j往i遍历
        while(i<j&&p<x[j])j--;
        if(i<j)
        {
            x[i]=x[j];
            i++;
        }
        //从i往j遍历
        while(i<j&&x[i]<p)i++;
        if(i<j)
        {
            x[j]=x[i];
            j--; 
        }
        //i==j
        x[i]=p;
        Qsort(x,l,i-1);
        Qsort(x,i+1,r);
    }
}
int main()
{
    int t;
    cin>>t;
    int  n;
    cin>>n;
    vector<int> x;
    while(t--)
    {
        int a;
        for(int i=0;i<n;i++)
        {
            cin>>a;
            x.push_back(a);
        }
        Qsort(x,0,n-1);
        for(int it:x)cout<<it<<' ';
        cout<<endl;
    }
    return 0;
}
